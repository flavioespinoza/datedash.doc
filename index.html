let any_date = 'Jan 7, 2019'
 
_d.date(any_date, '/')
// => 01/07/2019
 
_d.date(any_date, '-')
// => 01-07-2019
 
_d.date(any_date, '.')
// => 01.07.2019
 
_d.date(any_date, 'MMM DD YYYY')
// => Jan 07 2019
 
_d.date(any_date, 'england')
// => 07 Jan 2019
 
_d.date(any_date, 'uk')
// => 07 Jan 2019
 
_d.date(any_date, 'full')
// => Mon Jan 07 2019 00:00:00 GMT-0700 (Mountain Standard Time)

Installation Balls

{% raw %} 
// Using yarn
yarn add datedash
 
// Using npm
npm i datedash
{% endraw %}

Import ES6 Modules using esm or TypeScript

{% raw %} 
import _d, { addDays, subtractDays } from 'datedash'
 
_d.date('Mar 14, 2019', '/')
// => 3/14/2019
 
addDays('Mar 14, 2019', 1, '-')
// => 03-14-2019
 
subtractDays('Mar 14, 2019', 1, '-')
// => 03-14-2019
{% endraw %}

Require

{% raw %} 
const _d = require('datedash')
 
_d.date('Mar 14, 2019', '/')
// => 3/14/2019
 
_d.addDays('Mar 14, 2019', 1, '-')
// => 03-14-2019
 
_d.subtractDays('Mar 14, 2019', 1, '-')
// => 03-14-2019
{% endraw %}
